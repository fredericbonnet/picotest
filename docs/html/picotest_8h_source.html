532 _PICOTEST_TEST_DECLARE(_testName, 0, NULL); \
533 static int _testName##_testCaseRunner(void); \
534 int _testName(const char *cond) { \
535 int fail = 0; \

540 cond); \

542 case PICOTEST_FILTER_PASS: \

546 default:; \

551#define _PICOTEST_CASE_RUNNER_BEGIN(_testName) \
552 static int _testName##_testCaseRunner() { \
553 int abort; \
554 jmp_buf failureEnv; \
555 jmp_buf *oldEnv = _picoTest_failureEnv; \
556 int fail, oldFail = _picoTest_fail; \
557 _picoTest_failureEnv = &failureEnv; \
558 _picoTest_fail = 0; \
559 PICOTEST_CASE_ENTER(_PICOTEST_STRINGIZE(_testName)); \
560 abort = setjmp(failureEnv);

562#define _PICOTEST_CASE_RUNNER_END(_testName) \
563 fail = _picoTest_fail; \
564 PICOTEST_CASE_LEAVE(_PICOTEST_STRINGIZE(_testName), fail); \
565 _picoTest_failureEnv = oldEnv; \
566 _picoTest_fail = oldFail; \
567 return fail; \

1395 _PICOTEST_SUITE_ENUMERATE_SUBTEST, __VA_ARGS__) NULL}; \

1400 PicoTestMetadata **subtest = _suiteName##_subtests; \
1401 int fail = 0; \
1402 PICOTEST_SUITE_ENTER(_PICOTEST_STRINGIZE(_suiteName), nb); \
1403 for (; *subtest; subtest++) { \
1404 const int index = (int)(subtest - _suiteName##_subtests); \
1405 int sfail = 0; \
1406 PICOTEST_SUITE_BEFORE_SUBTEST(_PICOTEST_STRINGIZE(_suiteName), nb, \

1410 PICOTEST_SUITE_AFTER_SUBTEST(_PICOTEST_STRINGIZE(_suiteName), nb, \
1411 fail, index, (*subtest)->name, \
1412 sfail); \
1413 } \
1414 PICOTEST_SUITE_LEAVE(_PICOTEST_STRINGIZE(_suiteName), nb, fail); \
1415 return fail; \
1416 } \
